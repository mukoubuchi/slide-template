%% Definition slide: term intro / definition block / element list / point
%% (this section demonstrates the math environments)
\begin{frame}[label=dmproblem]
  \frametitle{Decision Making Problem}
  \begin{itemize}
    \item \bc{Decision Making Problem} is defined as a three-tuple
  \end{itemize}
  \begin{block}{Decision Making Problem}
    $D \coloneqq (A, \varTheta, c)$
  \end{block}
  \begin{itemize}
    \item[]
    \begin{itemize}
      \item $A$: \bc{Alternative}. A set of possible choices
      \item $\varTheta$: \bc{State of Nature}. A set of possible events in the future
      \item $c$: \bc{consequence}. A function defined as follows:
        $A \times \varTheta \xrightarrow{c} \mathbb{R};\, (a, \theta) \xmapsto{c} c\,(a, \theta)$
    \end{itemize}
  \end{itemize}
  \begin{alertblock}{Point}
    We can change $A$, but cannot $\varTheta$.
  \end{alertblock}
\end{frame}

%% Procedure slide: numbered steps / point
\begin{frame}[label=dmprocedure]
  \frametitle{Decision Making Procedure}
  \begin{enumerate}
    \item \bc{Goal Setting}: clarify goals of the decision making in order to
      evaluate preferences among alternatives
    \item \bc{Criteria Selection}: choose the best criterion so as to select
      the best alternative among alternatives
  \end{enumerate}
  \begin{alertblock}{Point}
    The preference of each $a \in A$ varies with goal.
  \end{alertblock}
\end{frame}
