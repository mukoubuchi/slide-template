\begin{frame}[label=dmproblem]
  \frametitle{意思決定問題}
  \begin{itemize}
    \item \BC{意思決定問題}は 3 つ組として定義される
  \end{itemize}
  \begin{block}{意思決定問題 (Decision Making Problem)}
    $D \coloneqq (A, \varTheta, c)$
  \end{block}
  \begin{itemize}
    \item[]
    \begin{itemize}
      \item $A$: \BC{選択肢} (Alternative)。選び得る選択の集合
      \item $\varTheta$: \BC{自然の状態} (State of Nature)。将来起こり得る事象の集合
      \item $c$: \BC{結果} (consequence)。次で定義される関数:
        $A \times \varTheta \xrightarrow{c} \mathbb{R};\, (a, \theta) \xmapsto{c} c\,(a, \theta)$
    \end{itemize}
  \end{itemize}
  \begin{alertblock}{Point}
    $A$ は変えられるが $\varTheta$ は変えられない。
  \end{alertblock}
\end{frame}

%% 手順スライドの例: 番号付きステップ / ポイント
\begin{frame}[label=dmprocedure]
  \frametitle{意思決定の手順}
  \begin{enumerate}
    \item \BC{目標設定} (Goal Setting): 選択肢間の選好を評価できるよう
      意思決定の目標を明確にする
    \item \BC{基準選択} (Criteria Selection): 最良の選択肢を選べるよう
      最良の基準を選択する
  \end{enumerate}
  \begin{alertblock}{Point}
    各選択肢 $a \in A$ の選好は目標によって変わる。
  \end{alertblock}
\end{frame}
